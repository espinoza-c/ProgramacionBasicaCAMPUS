static public void GreetCountries(string inputFile, string outputFile)
        {
            FileStream ostrm;
            StreamWriter writer;
            TextWriter oldOut = Console.Out;
            
            //Create file 
            ostrm = new FileStream(outputFile, FileMode.OpenOrCreate, FileAccess.Write);
            //Implement to write on file
            writer = new StreamWriter(ostrm);
           
            
            Console.SetOut(writer); //Todo lo que se imprima en la consola, se manda al objeto(nuestro nuevo archivo) writter
            String line;

            try
            {
                // Pass the file path and file name to the StreamReader constructor
                StreamReader sr = new StreamReader(inputFile);
                // Read the first line of text
                line = sr.ReadLine();

                while (line != null)
                {
                    //write the line to console window
                    Console.WriteLine("Saludos hasta " + line + "!.");
                    //read the next line
                    line = sr.ReadLine();
                }
                //close the file
                sr.Close();
                
            }
            catch (Exception e)
            {
                Console.WriteLine("Exception: " + e.Message);
            }
            Console.SetOut(oldOut);
            writer.Close();
            ostrm.Close();
            
            Console.WriteLine("Done");
        }
